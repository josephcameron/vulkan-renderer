project("Unit-Tests")
hunter_add_package(Catch)

    ${CMAKE_CURRENT_SOURCE_DIR}/main.cpp
)

)

    Catch2::Catch
)

add_test(${PROJECT_NAME} ${PROJECT_NAME})
